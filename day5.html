printing 13 table Using For Loop

        for (let i = 0; i <= 10; i++) {  // ( initialization, condition,  increment or decrement)
            console.log(13 + " X "+ i + " = " + 13*i);

While Loop

        var i=1; // initialization
        while(i<=10){    // condition
            
            console.log(12+ ' X '+i +' = ' + 12*i);
            i++;  //  increment or decrement
        }

        do{
            console.log(i);

        }
